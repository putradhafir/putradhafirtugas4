Pengantar Polimorfisme

Dart menyediakan beberapa mekanisme inisialisasi objek untuk berbagai kebutuhan arsitektur:

  1. Parameterized Constructor: ClassName(this.param)
  2. Contoh: Semua Kucing , Anjing , Burung adalah Hewan
  3. i.Compile-time polymorphism: Method overloading (tidak didukung Dart)
  4. ii. Runtime polymorphism: Method overriding (didukung Dart)

Pengantar Abstraksi

 Dalam konsepPemrograman Berorientasi Objek (OOP), 
      Abstraksi adalah proses penyembunyian detail 
      implementasi yang rumit dan hanya menampilkan 
      fungsi-fungsi penting saja kepada pengguna.

3. Class Abstract

Dapat memiliki method abstract (tanpa implementasi) dan concrete (dengan implementasi) 

// Nama: Putra Dhafir Nashrullah
// Nim: 251410009
// Kelas: SI2A

abstract class Kendaraan {
  String? merk;  // Field concrete
  
  void start();  // Method abstract
  
  void stop() {  // Method concrete
    print('Kendaraan berhenti');
  }
}

class Mobil extends Kendaraan {
  @override
  void start() {
    print('Mobil ${merk ?? 'tanpa merk'} menyala');
  }
  
  // Optional: override method stop jika ingin perilaku berbeda
  @override
  void stop() {
    print('Mobil ${merk ?? 'tanpa merk'} berhenti');
  }
}

void main() {
  Mobil mobil1 = Mobil();
  mobil1.merk = 'Toyota';
  mobil1.start();  // Output: Mobil Toyota menyala
  mobil1.stop();   // Output: Mobil Toyota berhenti
  
  Mobil mobil2 = Mobil();
  mobil2.start();  // Output: Mobil tanpa merk menyala
  mobil2.stop();   // Output: Mobil tanpa merk berhenti
}

5. Interface 

  Interface adalah sebuah "kontrak" atau cetak biru yang 
  menetapkan kemampuan apa saja yang harus dimiliki oleh sebuah kelas.
  Jika Abstraksi bicara tentang "menyembunyikan detail", maka Interface bicara tentang "standarisasi".

abstract interface class Flyable {
  void terbang(); 
}

class Bird implements Flyable {
  @override
  void terbang() => print('Burung terbang tinggi');
}

class Plane implements Flyable {
  @override
  void terbang() => print('Pesawat terbang cepat');
}

void main() {
  Bird burung = Bird();
  Plane pesawat = Plane();
  
  burung.terbang();   // Output: Burung terbang tinggi
  pesawat.terbang();  // Output: Pesawat terbang cepat
}

Perbedaan extends vs implements

Gunakan extends jika Anda ingin membuat kelas baru yang merupakan "anak" dari kelas lain Anda mendapatkan semua fiturnya secara gratis (warisan). Gunakan implements jika Anda ingin kelas Anda memiliki "kemampuan" tertentu (seperti printer yang bisa scan, atau mobil yang bisa terbang), tanpa peduli siapa orang tuanya.

// Nama: Putra Dhafir Nashrullah

abstract class Bentuk {
  double hitungLuas();
  double hitungKeliling();
  
  void info() {
    print('Luas: ${hitungLuas()}, Keliling: ${hitungKeliling()}');
  }
}

class Persegi implements Bentuk {
  double sisi;
  
  Persegi(this.sisi);
  
  @override
  double hitungLuas() => sisi * sisi;
  
  @override
  double hitungKeliling() => 4 * sisi;
  
  @override
  void info() {
    print('Persegi dengan sisi $sisi memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}');
  }
}

class Lingkaran implements Bentuk {
  double radius;
  
  Lingkaran(this.radius);
  
  @override
  double hitungLuas() => 3.14 * radius * radius;
  
  @override
  double hitungKeliling() => 2 * 3.14 * radius;
  
  @override
  void info() {
    print('Lingkaran dengan radius $radius memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}');
  }
}

void main() {
  Persegi persegi = Persegi(5);
  Lingkaran lingkaran = Lingkaran(7);
  
  persegi.info();
  lingkaran.info();
}

class MathUtils {
  static int factorial(int n) {
    if (n < 0) throw ArgumentError('Tidak bisa menghitung faktorial bilangan negatif');
    if (n == 0 || n == 1) return 1;
    int hasil = 1;
    for (int i = 2; i <= n; i++) {
      hasil *= i;
    }
    return hasil;
  }
  
  static bool isPrima(int n) {
    if (n <= 1) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (int i = 3; i * i <= n; i += 2) {
      if (n % i == 0) return false;
    }
    return true;
  }
  
  static double roundToDecimal(double value, int decimalPlaces) {
    double faktor = 1;
    for (int i = 0; i < decimalPlaces; i++) {
      faktor *= 10;
    }
    return (value * faktor).round() / faktor;
  }
  
  static int randomInt(int min, int max) {
    return min + (DateTime.now().millisecondsSinceEpoch % (max - min + 1));
  }
  
  static double randomDouble(double min, double max) {
    return min + (DateTime.now().millisecondsSinceEpoch % 10000) / 10000 * (max - min);
  }
  
  static double rataRata(List angka) {
    if (angka.isEmpty) return 0;
    double sum = 0;
    for (var a in angka) {
      sum += a;
    }
    return sum / angka.length;
  }
  
  static double median(List angka) {
    if (angka.isEmpty) return 0;
    List sorted = List.from(angka)..sort();
    int tengah = sorted.length ~/ 2;
    if (sorted.length % 2 == 0) {
      return (sorted[tengah - 1] + sorted[tengah]) / 2;
    } else {
      return sorted[tengah];
    }
  }
}

void main() {
  print('========== SOAL 5 ==========');
  
  print('Factorial 5: ${MathUtils.factorial(5)}');
  print('Factorial 7: ${MathUtils.factorial(7)}');
  print('Apakah 17 prima? ${MathUtils.isPrima(17)}');
  print('Apakah 20 prima? ${MathUtils.isPrima(20)}');
  print('3.14159 ke 2 desimal: ${MathUtils.roundToDecimal(3.14159, 2)}');
  print('Random Int (1-10): ${MathUtils.randomInt(1, 10)}');
  print('Random Double (0-1): ${MathUtils.randomDouble(0, 1)}');
  
  List nilai = [70, 85, 90, 75, 80, 95];
  print('List nilai: $nilai');
  print('Rata-rata: ${MathUtils.rataRata(nilai)}');
  print('Median: ${MathUtils.median(nilai)}');
}

abstract class MediaPlayer {
  void load(String file);
}

abstract class Playable {
  void play();
}

abstract class Pausable {
  void pause();
}

class AudioPlayer extends MediaPlayer implements Playable, Pausable {
  String? file;

  @override
  void load(String file) {
    this.file = file;
    print('Audio dimuat: $file');
  }

  @override
  void play() {
    print('Audio diputar');
  }

  @override
  void pause() {
    print('Audio dijeda');
  }
}

class VideoPlayer extends MediaPlayer implements Playable, Pausable {
  String? file;

  @override
  void load(String file) {
    this.file = file;
    print('Video dimuat: $file');
  }

  @override
  void play() {
    print('Video diputar');
  }

  @override
  void pause() {
    print('Video dijeda');
  }
}

void main() {
  List players = [
    AudioPlayer(),
    VideoPlayer(),
  ];

  for (var p in players) {
    p.load('file.mp3 / file.mp4');
  }

  if (p is Playable) {
    p.play();
  }

  if (p is Pausable) {
    p.pause();
  }

  print('---');
}
