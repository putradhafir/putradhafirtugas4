2.Pengantar Polimorfisme

Dart menyediakan beberapa mekanisme inisialisasi objek untuk berbagai kebutuhan arsitektur:

  1. Parameterized Constructor: ClassName(this.param)
  2. Contoh: Semua Kucing , Anjing , Burung adalah Hewan
  3. i.Compile-time polymorphism: Method overloading (tidak didukung Dart)
  4. ii. Runtime polymorphism: Method overriding (didukung Dart)
// Nama: Putra Dhafir Nashrullah
// Nim: 251410009
// Kelas: SI2A


class Bentuk {
void gambar() => print('Menggambar bentuk');
}
class Lingkaran extends Bentuk {
@override
void gambar() => print('Menggambar lingkaran');
}
class Persegi extends Bentuk {
@override
void gambar() => print('Menggambar persegi');
}
void main() {
Bentuk bentuk1 = Lingkaran();  // Polimorfisme
Bentuk bentuk2 = Persegi();    
// Polimorfisme
bentuk1.gambar();  // Output: Menggambar lingkaran
bentuk2.gambar();  // Output: Menggambar persegi

}

7. Interface

  Interface adalah sebuah "kontrak" atau cetak biru yang 
  menetapkan kemampuan apa saja yang harus dimiliki oleh sebuah kelas.
  Jika Abstraksi bicara tentang "menyembunyikan detail", maka Interface bicara tentang "standarisasi".
        

// Nama: Putra Dhafir Nashrullah
// Nim: 251410009
// Kelas: SI2A

interface class Flyable {
  void terbang() => print('Terbang...');
}

class KupuKupu implements Flyable {
  @override
  void terbang() => print('Kupu-kupu terbang indah');
}

class Drone implements Flyable {
  @override
  void terbang() => print('Drone terbang cepat');
}

void main() {
  KupuKupu k = KupuKupu();
  Drone d = Drone();

  k.terbang();
  d.terbang();
}

Perbedaan extends vs implements

✅ Gunakan extends jika Anda ingin membuat kelas baru yang merupakan "anak" dari kelas lain. Anda mendapatkan semua fiturnya secara gratis (warisan).

⚙️ Gunakan implements jika Anda ingin kelas Anda memiliki "kemampuan" tertentu (seperti printer yang bisa scan, atau mobil yang bisa terbang), tanpa peduli siapa orang tuanya.

10.Sistem Bentuk Geometri

Menggabungkan abstraksi dan polimorfisme. Kelas Bentuk mendefinisikan metode hitungLuas() dan hitungKeliling(). Persegi dan Lingkaran mengimplementasikan rumusnya masing-masing 

// Nama: Putra Dhafir Nashrullah
// Nim: 251410009
// Kelas: SI2A

abstract class Bentuk {
  double hitungLuas();
  double hitungKeliling();
  
  void info() {
    print('Luas: ${hitungLuas()}, Keliling: ${hitungKeliling()}');
  }
}

class Persegi implements Bentuk {
  double sisi;
  
  Persegi(this.sisi);
  
  @override
  double hitungLuas() => sisi * sisi;
  
  @override
  double hitungKeliling() => 4 * sisi;
  
  @override
  void info() {
    print('Persegi dengan sisi $sisi memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}');
  }
}

class Lingkaran implements Bentuk {
  double radius;
  
  Lingkaran(this.radius);
  
  @override
  double hitungLuas() => 3.14 * radius * radius;
  
  @override
  double hitungKeliling() => 2 * 3.14 * radius;
  
  @override
  void info() {
    print('Lingkaran dengan radius $radius memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}');
  }
}

void main() {
  Persegi persegi = Persegi(5);
  Lingkaran lingkaran = Lingkaran(7);
  
  persegi.info();
  lingkaran.info();
}

import 'dart:math';

class MathUtils {
  static int factorial(int n) => n <= 1 ? 1 : n * factorial(n - 1);

  static bool isPrime(int n) {
    if (n < 2) return false;
    // Perbaikan: sqrt membutuhkan double, lalu konversi balik ke int
    int limit = sqrt(n.toDouble()).toInt();
    for (int i = 2; i <= limit; i++) 
      if (n % i == 0) return false;
    return true;
  }

  static double roundDouble(double val, int places) =>
    double.parse(val.toStringAsFixed(places));

  static int randomInt(int min, int max) => 
    min + Random().nextInt(max - min + 1);

  // PERBAIKAN: tambahkan tipe data spesifik
  static double average(List nums) => 
    nums.isEmpty ? 0 : nums.reduce((a, b) => a + b) / nums.length;

  // PERBAIKAN: tambahkan tipe data spesifik
  static double median(List nums) {
    if (nums.isEmpty) return 0;
    var s = List.from(nums)..sort();
    int m = s.length ~/ 2;
    return s.length % 2 != 0 
        ? s[m].toDouble() 
        : (s[m - 1] + s[m]) / 2.0;
  }
}

// Contoh penggunaan
void main() {
  print('Factorial 5: ${MathUtils.factorial(5)}');
  print('Is 17 prime? ${MathUtils.isPrime(17)}');
  print('Round 3.14159 to 2 decimals: ${MathUtils.roundDouble(3.14159, 2)}');
  print('Random int (1-10): ${MathUtils.randomInt(1, 10)}');
  
  List nilai = [70, 85, 90, 75, 80, 95];
  print('Average: ${MathUtils.average(nilai)}');
  print('Median: ${MathUtils.median(nilai)}');
}

  void play() => print('Memutar Audio MP3...');
}

  @override

  @override
  void play() => print('Memutar Live Stream...');
